    /**
 * Will plot points depending on what is defined as the mix attribute,
 * and will sample (w/o replacement) from the dataset to display points
 * with all defined values for the mix attribute.
 *
 * E.g. if properties["mix_data"] = {"point_size" = [1, 2, 3, 4]}
 *      and there are 100 points in the dataset, will plot 25 points with
 *      point_size = 1, 25 points with point_size = 2, etc...
 * 
 * E.g. if properties["mix_data"] = {"point_size" = [1, 2], "point_color" = [black, blue]}
 *      and there are 100 points in the dataset, will first convert the set into the array 
 *      [{"point_size" = 1, "point_color" = red}, {"point_size" = 2, "point_color" = red},
 *      {"point_size" = 1, "point_color" = blue}, {"point_size" = 2, "point_color" = blue}], 
 *      then plot 25 points with point_size = 1 and point_color = red, 
 *      25 points with point_size = 2 and point_color = red, etc...
 *
 * E.g. if properties["mix_data"] = {"point_size" = [1, 2], "point_color" = [black, blue]}
 *      and there are 100 points in the dataset, will first convert the set into the array 
 *      [{"point_size" = 1, "point_color" = red}, {"point_size" = 2, "point_color" = red},
 *      {"point_size" = 1, "point_color" = blue}, {"point_size" = 2, "point_color" = blue}], 
 *      then plot 25 points with point_size = 1 and point_color = red, 
 *      25 points with point_size = 2 and point_color = red, etc...
 *
 * @param {object}         chart
 * @param {function}       xscale
 * @param {function}       yscale
 * @param {array}          dataset ([x_value, y_value])
 * @param {assoc. array}   e.g. {"point_shape": "square", "point_size": 5 .... }
 */
export function prepare_mix_graph(chart, xscale, yscale, dataset, properties, plot_graph) {

    let NUM_GROUPS = 1;
    let mix_attribs = []
    let mix_attribs_pairs = [{}];
  
    for (let x = 0; x < Object.keys(properties["mix_data"]).length; x++){
      let mix_attrib = Object.keys(properties["mix_data"])[x];
      mix_attribs.push(mix_attrib);
      NUM_GROUPS = NUM_GROUPS * properties["mix_data"][mix_attrib].length;
  
      let new_mix_attribs = [];
      properties["mix_data"][mix_attrib].forEach(e => {
  
        let temp_mix_attribs = [];
        mix_attribs_pairs.forEach(original => {
          var copy = {};
          for(var key in original)
            {
                copy[key] = original[key];
            }
          temp_mix_attribs.push(copy)
        });
  
        temp_mix_attribs.forEach(element => {
          element[mix_attrib] = e;
        });
        new_mix_attribs = new_mix_attribs.concat(temp_mix_attribs);  
      });
      mix_attribs_pairs = new_mix_attribs;
    }
  
    let POINTS_PER_GROUP = dataset.length / NUM_GROUPS;
  
    if(dataset.length % NUM_GROUPS > 0){
      console.log("ERROR: DATASET CAN'T BE DEVIDED BY NUM_GROUPS EVENLY")
      console.log("dataset.length:")
      console.log(dataset.length);
      console.log("NUM_GROUPS:")
      console.log(NUM_GROUPS);
      console.log("POINTS_PER_GROUP:")
      console.log(POINTS_PER_GROUP);
      console.log("dataset.length % NUM_GROUPS = " + (dataset.length % NUM_GROUPS));
    }
  
  
    for (let i = 0; i < NUM_GROUPS; i++) {
  
      let grouped_dataset = [];
  
      for (let j = 0; j < POINTS_PER_GROUP; j++) {
  
        let rand_index = Math.floor(Math.random() * dataset.length);
        grouped_dataset.push(dataset[rand_index]);
  
        // Remove from dataset
        dataset.splice(rand_index, 1);
      }
  
      // Override the property if it is a mix attribute
    
      for (let y = 0; y < mix_attribs.length; y++) {
        properties[mix_attribs[y]] = mix_attribs_pairs[i][mix_attribs[y]];         
      }
      // Plot this group of points with the ith value for the mix attribute
      plot_graph(chart, xscale, yscale, grouped_dataset, properties);
    }
  }
  
  /**
   * The same as prepare_mix_graph but now the first item in the mix by attributes array
   * is treated as the target
   * The user can specify how many target points there are
   * The rest of the mix by attributes are treated as distractors
   * and the remaining number of points is distributed evenly among them
   * 
   * e.g. if if properties["mix_by_attribute_targeted"] = {"point_size" = [1, 2, 3, 4], num_target: 10}
   *      and there are 100 points in the dataset, will plot 10 points with
   *      point_size = 1, 30 points with point_size = 2, 3 and 4.
   *
   * @param {object}         chart
   * @param {function}       xscale
   * @param {function}       yscale
   * @param {array}          dataset ([x_value, y_value])
   * @param {assoc. array}   e.g. {"point_shape": "square", "point_size": 5 .... }
   */
  export function prepare_targeted_mix_graph(chart, xscale, yscale, dataset, properties) {
  
    let mix_attrib = Object.keys(properties["mix_data"])[0];
    let POINTS_FOR_TARGET = properties["mix_data"]["num_target"];
    
    let NUM_GROUPS = properties["mix_data"][mix_attrib].length;
    if (POINTS_FOR_TARGET < 0) {
      NUM_GROUPS = NUM_GROUPS - 1;
    }
    let POINTS_PER_GROUP = (dataset.length - POINTS_FOR_TARGET) / NUM_GROUPS;
  
    for (let i = 0; i < NUM_GROUPS; i++) {
  
      let grouped_dataset = [];
      if (i == 0 && POINTS_FOR_TARGET == 0) {
        continue;
      }
    
      if (i == 0) {
        for (let h = 0; h < POINTS_FOR_TARGET; h++) {
          let rand_index = Math.floor(Math.random() * dataset.length);
          grouped_dataset.push(dataset[rand_index]);
          // Remove from dataset
          dataset.splice(rand_index, 1);
        }
      } else {
          for (let j = 0; j < POINTS_PER_GROUP; j++) {
            let rand_index = Math.floor(Math.random() * dataset.length);
            grouped_dataset.push(dataset[rand_index]);
  
            // Remove from dataset
            dataset.splice(rand_index, 1);
          }
      }
  
      // Override the property if it is a mix attribute
      properties[mix_attrib] = properties["mix_data"][mix_attrib][i];
      // Plot this group of points with the ith value for the mix attribute
      plot_graph(chart, xscale, yscale, grouped_dataset, properties);
    }
  
  }
  
  /**
   * similar to mix_by_attribute, prepare_array_mix_graph will plot points
   * depending on what is defined as the mix attributes. However, mix_by_attribute_array
   * allows for several attributes to be changed within a single graph.
   * 
   * e.g. if properties["mix_by_attribute_array"] =
   *      [{"point_size" = 1}, 
   *      {"point_size" = 2, "point_color" = blue, "point_shape" = circle}, 
   *      {"point_size" = 3, "point_color" = red},
   *      {"point_size" = 4, "point_color" = red},]
   *      and there are 100 points in the dataset, will plot 25 points with
   *      point_size = 1, 25 points with point_size = 2 and point_color = blue and point_shape = circle, etc...
   * 
   * 
   * @param {*} chart 
   * @param {*} xscale 
   * @param {*} yscale 
   * @param {*} dataset ([x_value, y_value])
   * @param {*} properties {"point_shape": "square", "point_size": 5 .... }
   */
  export function prepare_array_mix_graph(chart, xscale, yscale, dataset, properties, plot_graph) {
    let NUM_GROUPS = properties["mix_data"].length;
    let mix_attrib_array = properties["mix_data"];
    let POINTS_PER_GROUP = dataset.length / NUM_GROUPS;
  
    for (let i = 0; i < NUM_GROUPS; i++) {
  
      let grouped_dataset = [];
  
      for (let j = 0; j < POINTS_PER_GROUP; j++) {
        //make sure the dataset is not empty
        if(dataset.length > 0){
          let rand_index = Math.floor(Math.random() * dataset.length);
          grouped_dataset.push(dataset[rand_index]);
  
          // Remove from dataset
          dataset.splice(rand_index, 1);
        }
      }
  
      // Override the property if it is a mix attribute
      for(var attrib in mix_attrib_array[i]){
        // console.log("attrib")
        // console.log(attrib);
        properties[attrib] = mix_attrib_array[i][attrib];   
      }    
  
      // console.log("properties")
      // console.log(properties);
  
      // Plot this group of points with the ith value for the mix attribute
      plot_graph(chart, xscale, yscale, grouped_dataset, properties);
    }
  }
