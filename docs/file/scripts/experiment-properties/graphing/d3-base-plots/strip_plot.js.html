/**
 * D3 code to plot a strip plot
 * @param {array} dataset - Array of data coordinates to plot the strip plot lines
 * @param {assoc array} properties - D3 Graph Properties such as height, width, axis thickness, etc...
 * @param {assoc array} style_attributes - Optional style attributes for css styling
 */
function create_strip_plot(attributes, style_attributes = null, distractor_attributes = null) {
	let dataset = attributes.dataset;
    let properties = attributes.graph_attributes;
	let width = getWidth(properties);
    let height = getHeight(properties);

	let xscale_data = getXScaleData(properties,width);
	const yscale_data = getYDataScale(properties);

    const xscale_x_axis = d3.scaleLinear()
                   .domain([0, 1]) 
                   .range([0, width]);

    // Create axes
    const x_axis = d3.axisBottom()
                   .scale(xscale_x_axis)
                   .tickSize(0);

	const chart = setStyleAttributes(style_attributes,height,width)
	
	chart.append("g")
	    .attr("transform", "translate(0, " + height/2  +")") 
	    .call(x_axis)
	
	add_graph_features(chart, attributes, distractor_attributes, xscale_data, yscale_data, plotStripYValue);

	// Plot main feature representation
	//plotStripYValue(chart, properties, dataset, height, xscale_data)

	// Plot Distractor feature representation
	plotStripZValue(chart, properties, height, xscale_data)

	// Set axis color
	chart.selectAll("path")
	   .attr("stroke", properties["axis_color"])

	// Remove tick labels
	chart.selectAll("text").remove();   

}

function plotStripYValue(chart, xscale_data, y_data_scale, dataset, properties){
		let height = getHeight(properties);
		switch(properties["y_value_representation"]){
			case "orientation":
				const rotate_data = d3.scaleLinear()
									.domain([0, 1])
									.range([-45, 45]);

				chart.selectAll("strip")
					.data(dataset)
					.enter()
					.append("rect")
					.attr("x", function(d){
						return xscale_data(d[0]) - properties.strip_width/2;
					})
					.attr("y", 0)
					.attr("width", properties.strip_width)
					.attr("height", function(d){
						let angle_deg = y_data_scale(d[1]);
						let extra_height = getExtraHeight(angle_deg, height);
						return height + 2*extra_height;
					})
					.attr("transform", function (d) {
						let cx = xscale_data(d[0]);
						let angle_deg = y_data_scale(d[1]);
						let extra_height = getExtraHeight(angle_deg, height);
						let cy = height/2;
						return `translate(0,${-extra_height}) rotate(${angle_deg},${cx},${cy})`
					})
					.attr('fill', properties["fill_color"])
				break;	
			case "luminance":
				const luminance_scale = d3.scaleLinear()
										.domain([0, 1])
										.range([0.9, 0])

				chart.selectAll("strip")
					.data(dataset)
					.enter()
					.append("rect")
					.attr("x", function(d){
						return xscale_data(d[0]) - properties.strip_width/2;
					})
					.attr("y", 0)
					.attr("height", height)
					.attr("width", properties.strip_width)
					.attr("fill", function(d){	// luminance properties
						let hsl_color = new d3.hsl(1,0,y_data_scale(d[1]))
						return hsl_color
					})
				break;
			case "height":
				const yscale = d3.scaleLinear()
									.domain([0, 1])
									.range([height, 0]);

				chart.selectAll("strip")
					.data(dataset)
					.enter()
					.append("rect")
					.attr("x", function(d){
						return xscale_data(d[0]) - properties.strip_width/2;
					})
					.attr("y", function(d){
						let diff = height/2 - yscale(d[1])/2
						return diff
					})
					.attr("height", function(d){
						return y_data_scale(d[1])
					})
					.attr("width", properties.strip_width)
					.attr('fill', properties["fill_color"])
				break;
			case "cielab":
				const cielab_scales = generateCielablScales(properties);
				//console.log(cielab_scales)
				chart.selectAll("strip")
					.data(dataset)
					.enter()
					.append("rect")
					.attr("x", function(d){
						return xscale_data(d[0]) - properties.strip_width/2;
					})
					.attr("y", 0)
					.attr("height", height)
					.attr("width", properties.strip_width)
					.attr("fill", function(d){	// luminance properties
						let cielab_color = new d3.lab(cielab_scales.l_range(d[1]),cielab_scales.a_range(d[1]),cielab_scales.b_range(d[1]), 1.0);
						//console.log(cielab_color)
						return cielab_color
					})
					break;
			default:
				throw Error(`Representation ${properties["y_value_representation"]} has not been defined in the strip plot generator.`);
				break;
		}
	
}

/**
 * Plots the z-value features of the strip plot by modifying the existing rect object attributes
 * @param {object} chart 
 * @param {assoc array} properties 
 * @param {int} height 
 * @param {function} xscale_data 
 * @param {function} rotate_data 
 * @param {function} yscale 
 */
function plotStripZValue(chart, properties, height, xscale_data, rotate_data, yscale, luminance_scale){
	switch(properties["distractor_value_feature"]){
		case "orientation":
			const rotate_data = d3.scaleLinear()
									.domain([0, 1])
									.range([-45, 45]);
			chart.selectAll("rect")
				.attr("height", function(d){
					let angle_deg = rotate_data(d[2]);
					let extra_height = getExtraHeight(angle_deg, height);
					return height + 2*extra_height;
				})
				.attr("transform", function (d) {
					let cx = xscale_data(d[0]);
					let angle_deg = rotate_data(d[2]);
					let extra_height = getExtraHeight(angle_deg, height);
					let cy = height/2;
					return `translate(0,${-extra_height}) rotate(${angle_deg},${cx},${cy})`
				})
				break;
		case "luminance":
			const luminance_scale = d3.scaleLinear()
										.domain([0, 1])
										.range([0.9, 0])

			chart.selectAll("rect")
				.attr("fill", function(d){	// luminance properties
					let hsl_color = new d3.hsl(1,0,luminance_scale(d[2]))
					return hsl_color
				})
			break;
		case "height":
			const yscale = d3.scaleLinear()
									.domain([0, 1])
									.range([height, 0]);

			attr("y", function(d){
				let diff = height/2 - yscale(d[2])/2
				return diff
			})
			.attr("height", function(d){
				return yscale(d[2])
			})
			break;
		case "none":
			break
		default:
			break;
	}
}

/**
 * Gets the correct axis width with buffer if the properties has it
 * @param {asoc array} properties 
 * @returns {float}
 */
function getWidth(properties){
	let width = properties.axis_length;
	if(properties.buffer){
		width += properties.buffer;
	}
	return width;
}

/**
 * Gets the correct axis height. Generalized as a function for future proofing.
 * @param {asoc array} properties 
 * @returns {float}
 */
function getHeight(properties){
	let height = properties.line_height;

	return height;
}

function getXScaleData(properties, width){
	let xscale_data;
	let buffer = properties.buffer;
	if (properties.buffer){
		xscale_data = d3.scaleLinear()
						.domain([0, 1]) 
						.range([buffer, width - buffer]);
	}
	else{
		xscale_data = d3.scaleLinear()
                   		.domain([0, 1]) 
                   		.range([width * 0.05, width * 0.95]);
	}
	return xscale_data;
}

 *

 * @returns {function}
 */
function getYDataScale(properties){
	const min = properties["y_value_range"]["min"];
	const max = properties["y_value_range"]["max"];

	const yscale_data = d3.scaleLinear()
						.domain([0, 1])
						.range([min, max])
	return yscale_data;
}

/**
 * Gets the length of the rotated line so that the height of the graph remains constant.
 * Otherwise the height of the line decreases. Line needs to be offset by the extra height
 * @param {float} angle_deg angle of rotation in degrees (-45,45)
 * @param {float} height    target height of the graph
 * @returns {float} 		extra height so rotated line matches target height of graph 
 */
function getExtraHeight(angle_deg,height){
	//let angle = Math.abs(rotate_data(d[1])) * Math.PI / 180; // rads
	//let angle_deg = 45
	let angle_rad = angle_deg * Math.PI / 180; // rads
	let ninety = Math.PI/2;
	let short_height = height * Math.cos(angle_rad); // https://math.stackexchange.com/questions/260096/find-the-coordinates-of-a-point-on-a-circle
	//console.log(angle, short_height)
	let height_diff = height - short_height;
	let extra_height =  height_diff / Math.sin(ninety - angle_rad); // https://www.softschools.com/math/calculus/using_the_law_of_sines_to_find_an_unknown_side/
	//console.log(height / Math.cos(rotate_data(d[1]) * Math.PI / 180))
	//console.log (height, extra_height)
	return extra_height; 
}

function generateCielablScales(properties){
	if(properties.y_value_representation != "cielab"){
		return;
	}
	let l_min = properties.y_value_range.min.l;
	let a_min = properties.y_value_range.min.a;
	let b_min = properties.y_value_range.min.b;
	let l_max = properties.y_value_range.max.l;
	let a_max = properties.y_value_range.max.a;
	let b_max = properties.y_value_range.max.b;

	let cielab_scales = {};
	cielab_scales.l_range = d3.scaleLinear()
								.domain([0, 1]) 
								.range([l_min, l_max]);

	cielab_scales.a_range = d3.scaleLinear()
								.domain([0, 1]) 
								.range([a_min, a_max]);
	cielab_scales.b_range = d3.scaleLinear()
								.domain([0, 1]) 
								.range([b_min, b_max]);
	return cielab_scales;
}

/**
 * Translates and Rotates the strip plot line for orientation feature
 * First translates the y position of the line to account for the extra height from getHeightOrientation
 * Then rotates it along cx,cy by the given angle. cy is calculated internally
 * 
 * @param {float} cx 		center x coordinate for rotation
 * @param {float} angle_deg rotation angle in degrees
 * @param {float} height 	target height of the graph
 * @returns {string} 		d3 transformation function formated as a string
 */
function translateAndRotateOrientation(cx,angle_deg,height){
	//let angle_deg = 45
	//let angle_sign = Math.sign(angle_deg)
	let angle_rad = angle_deg * Math.PI / 180; // rads
	let ninety = Math.PI/2;
	let cos_a = Math.cos(angle_rad)
	let short_height = height * cos_a; // https://math.stackexchange.com/questions/260096/find-the-coordinates-of-a-point-on-a-circle
	//console.log(angle_rad, cos_a , short_height)
	let height_diff = height - short_height;
	let extra_height =  height_diff / Math.sin(ninety - angle_rad); // https://www.softschools.com/math/calculus/using_the_law_of_sines_to_find_an_unknown_side/
	let rotation_cy = height/2;
	// subtract the extra height (half the total additional height of the line) so that rotation cy aligns with y axis position
	// then translate line to the right at a ratio equal to 1 - the cosine (ie dont translate if) rotate by rotate_data angle, along cx and cy (half the height)
	//return `translate(0,${-extra_height}) rotate(${angle_deg},${cx},${rotation_cy})`
	return `rotate(${angle_deg},${cx},${rotation_cy})`

}
