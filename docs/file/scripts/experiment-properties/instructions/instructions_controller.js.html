    import { EXPERIMENTS } from "/config/experiments-config.js";
import { CONDITIONS } from "/config/conditions-config.js";
import { correlation_dependencies } from "/scripts/helpers/correlation_dependencies.js"
export { get_instructions, get_jspsych_instructions, generate_subcond_instruction_block }


/**
 * Generates a jsPsych instruction trial for the current condition if it has been defined for it, or
 * the default experiment instruction (which needs to be defined otherwise an error is thrown)
 * 
 * It handles the old instruction format where the instructions are a plain HTML string and image array that replaces
 * the image key words in the html string with get_html_with_images(). This is done through either entering no instruction name
 * (the jnd timeline does this), which defaults to the old behavior, or 
 * 
 * It also handles experiments with both type of instruction types such as new_sequencing by givin it the experiment name as
 * the instruction name. 
 * 
 * Otherwise (default) it gets the jsPsych trial instruction object with the given instruction_name
 * 
 * @param {object} experiment 
 * @param {string} instruction_name 
 * @param {boolean} custom_instructions 
 * @returns {jsPsych trial}
 */
function get_jspsych_instructions(experiment, instruction_name = "", custom_instructions = false) {
  let experiment_name = experiment.constructor.name.toLowerCase();
  let experiment_instructions = EXPERIMENTS[experiment_name].instructions;
  let condition_instructions_object = CONDITIONS[experiment.condition_name].instructions;

  switch (instruction_name) {
    case "":
      //check if there is specific images for the block
      if (is_block_specific_image(experiment) && condition_instructions_object) {
        return condition_custom_block_html_to_trial(experiment, experiment_instructions);
      }
      // Backwards compatibility case for experiments with only HTML instructions
      if (condition_instructions_object) {
        if (is_custom_html_trial_instruction(condition_instructions_object, experiment_name)) {
          return condition_custom_html_to_trial(condition_instructions_object[experiment_name], experiment_instructions)
        }
        throw new Error("Condition instructions with no name default to old HTML instruction format. Double check if this condition is formated as that, otherwise use the specific instruction name or")
      }
      if (experiment_instructions.default_html || experiment_instructions.default_images) {
        return experiment_custom_html_to_trial(experiment_instructions);
      }
      else {
        throw new Error("No default html defined in instructions object for this condition for experiment " + experiment.constructor.name);
      }
    case experiment_name:
      // Backwards compatibility case for experiments with combined instruction types
      if (condition_instructions_object) {
        if (condition_instructions_object[experiment_name]) {
          return condition_instructions_object[experiment_name];
        }
      }
      if (experiment_instructions.default_html || experiment_instructions.default_images) {
        return experiment_custom_html_to_trial(experiment_instructions);
      }
      else {
        throw new Error("No default html defined in instructions object for this condition for experiment " + experiment.constructor.name);
      }
    default:
      let instruction = instruction_name + "_html";
      if (condition_instructions_object && condition_instructions_object[instruction]) {

        return condition_instructions_object[instruction];
      }
      if (experiment_instructions[instruction]) {

        return experiment_instructions[instruction];
      }
      else {
        throw new Error("No jspsych html defined in instructions object for this condition for experiment " + experiment.constructor.name);
      }
  }
}

/**
 * Generates the instructions html for the given experiment.
 * Loads default Instruction object from experiments-config.js
 * and Condition Instruction object from conditions-config.js if it exists
 *
 * @param   {object}   experiment
 *
 * @return  {string}   html for the instructions jsPsych block
 */
function get_instructions(experiment) {

  // EXPERIMENTS from experiments-config.js
  let experiment_name = experiment.constructor.name.toLowerCase();
  let instructions_info = EXPERIMENTS[experiment_name].instructions;

  if (!instructions_info.default_html || instructions_info.default_html.length === 0) {
    throw new Error("No default html specified for experiment " + experiment_name);
  }

  if (!instructions_info.default_images || instructions_info.default_images.length === 0) {
    throw new Error("No default images are defined for experiment " + experiment_name);
  }

  let default_html = instructions_info.default_html;
  let default_images = instructions_info.default_images;
  let condition_instructions_object = CONDITIONS[experiment.condition_name].instructions;

  if (condition_instructions_object) {
    return get_custom_instructions(experiment, condition_instructions_object, default_html, default_images);
  }
  // Else use default html and images
  else {
    return get_html_with_images(default_html, default_images);
  }

}

/**
 * Generates custom instructions depending on whether custom_html is defined, or 
 * custom_images is defined.
 *
 * @param   {object}   experiment
 * @param   {object}   condition_instructions_object
 * @param   {string}   default_html
 * @param   {array}    default_images
 *
 * @return  {string}   html for custom instructions
 */
function get_custom_instructions(experiment, condition_instructions_object, default_html, default_images) {

  // Validation check that experiment in instructions object is a supported experiment for the condition
  Object.keys(condition_instructions_object).forEach(function (exp) {
    if (!CONDITIONS[experiment.condition_name]["experiment"].includes(exp)) {
      throw new Error("Experiment " + exp + " is listed within instructions key but is not among the supported experiments " +
        "listed for the condition. Listed experiments are: " + CONDITIONS[experiment.condition_name]["experiment"] + ".");
    }
  });

  let condition_instructions = condition_instructions_object[experiment.constructor.name.toLowerCase()];

  // If no instructions for this experiment, use default
  if (!condition_instructions) {
    return get_html_with_images(default_html, default_images);
  }
  else {

    // Validation check that must only have one key (custom_html or custom_images) defined
    if (Object.keys(condition_instructions).length > 1) {
      throw new Error("Only one key permitted inside instructions object for experiment " + experiment.constructor.name.toLowerCase() + ". Must be " +
        "either custom_html or custom_images.");
    }

    // If there is custom html
    if (condition_instructions.custom_html) {
      return condition_instructions.custom_html;
    }
    // If there is custom images
    else if (condition_instructions.custom_images) {
      let images = condition_instructions.custom_images;

      // Validation check that specified num of custom_images == num of default_images
      if (images.length !== default_images.length) {
        throw new Error("The default instruction for this experiment requires " + default_images.length + " image(s). " + images.length + " image(s) were specified for the condition instead.");
      }

      return get_html_with_images(default_html, images);
    } else {
      throw new Error("No custom_html or custom_images defined in instructions object for this condition for experiment " + experiment.constructor.name);
    }

  }
}

/**
 * Replaces any 'image1.png', 'image2.png' ... etc with each string 
 * entry in images.
 *
 * NOTE: When defining default_html in experiments-config.js, need to name
 *       images as 'image1.png', 'image2.png' so this code knows what to replace
 *
 * @param   {string}   html
 * @param   {array}    images 
 *
 * @return  {string}   html with any 'image1.png' etc replaced with entries in images
 */
function get_html_with_images(html, images) {

  let counter = 1;
  console.log(images);
  for (let image_url of images) {
    let url_to_replace = "image" + counter + ".png";

    html = html.replace(url_to_replace, image_url);

    counter++;
  }

  return html;
}



/**
 * Transforms legacy HTML instructions in conditions-config to jsPsych trials.
 * They are html-keyboard-response trials to retain "press any key to continue"
 * functionality rather than use the "next page" mouse button.
 * 
 * @param {string} experiment
 * @param {assoc array} html_instructions 
 * @return {jsPsych trial}
 */
function condition_custom_html_to_trial(condition_object, experiment_instructions) {
  let custom_html = experiment_instructions.default_html;
  if (condition_object) {
    if (condition_object.custom_html && condition_object.custom_images) {
      custom_html = get_html_with_images(condition_object.custom_html, condition_object.custom_images)
    }
    else if (condition_object.custom_images) {
      custom_html = get_html_with_images(experiment_instructions.default_html, condition_object.custom_images)
    }
    else {
      custom_html = condition_object.custom_html;
    }
  }

  let instruction_trial = {
    type: "html-keyboard-response",
    data: { screen_id: "instruction", },
    stimulus: custom_html,
  }
  return instruction_trial
}

/**
 * Transforms legacy HTML instructions and custom block images in conditions-config to jsPsych trials.
 * They are html-keyboard-response trials to retain "press any key to continue"
 * functionality rather than use the "next page" mouse button.
 * 
 * @param {string} experiment (in conditions-config)
 * @param {assoc array} html_instructions 
 * @returns {jsPsych trial}
 */
function condition_custom_block_html_to_trial(experiment, experiment_instructions) {
  let first_block = get_first_block_name(experiment);
  let experiment_name = experiment.constructor.name.toLowerCase();
  if (experiment_name == "hybrid") {
    experiment_name = experiment.current_experiment;
  }


  //get the custom block image paths
  let condition_instructions = CONDITIONS[experiment.condition_name].instructions[experiment_name];
  let block_images = condition_instructions.custom_block_images[`block_${first_block}_images`];

  //generate default trial without custom images
  let custom_html = experiment_instructions.default_html;
  if (condition_instructions.custom_html) {
    custom_html = get_html_with_images(condition_instructions.custom_html, block_images)
  }
  else {
    custom_html = get_html_with_images(experiment_instructions.default_html, block_images)
  }

  let trial = {
    type: "html-keyboard-response",
    data: { screen_id: "instruction", },
    stimulus: custom_html,
  }

  return trial;
}

/**
 * Generates the experiment's instruction blocks.
 * @param {object} experiment 
 * @param {string} block_type
 * @returns {trial}
 */
function generate_subcond_instruction_block(experiment, block_type) {
  let instruction_trial = {
    type: "instructions",
    data: {},
    key_forward: " ",
    pages: [],
    on_start: function (trial) {
      console.log("test")
      let index = correlation_dependencies.get_subcondition_index(experiment, block_type);
      let constants = correlation_dependencies.get_constants(experiment, block_type, index);
      let curr_block = constants.block;
      let experiment_name = experiment.constructor.name.toLowerCase();
      let current_instruction = get_subcondition_instruction_trial(experiment, experiment_name, curr_block);
      let curr_pages = current_instruction.pages;
      trial.pages = curr_pages;
      trial.data = { screen_id: `${experiment_name}_${curr_block}_html` }
    }
  };
  let subcond_instruction_block = {
    timeline: [instruction_trial],
    conditional_function: function () {
      let index = correlation_dependencies.get_subcondition_index(experiment, block_type);
      let constants = correlation_dependencies.get_constants(experiment, block_type, index);
      return ((constants.block != undefined) && constants.subcondition_instructions === true &&
        correlation_dependencies.is_start_of_subcondition(experiment, block_type) &&
        correlation_dependencies.is_start_of_block(experiment, block_type))
    },
  }
  return subcond_instruction_block;
}

/**
 * Gets the instruction trial defined for the current block/sub condition in the condition.
 * @param {object} experiment 
 * @returns {trial}
 */
function get_subcondition_instruction_trial(experiment, experiment_name, curr_block) {
  let block_instruction_html_key = `${experiment_name}_${curr_block}_html`;
  let block_instruction_image_key = `${experiment_name}_${curr_block}_images`;
  let block_instructions_text = null;
  let block_instructions_images = null;
  console.log(block_instruction_html_key, block_instruction_image_key)

  // Get jsPsych trial 
  if (CONDITIONS[experiment.condition_name]["block_instructions"]) {
    if (CONDITIONS[experiment.condition_name]["block_instructions"][block_instruction_html_key]) {
      block_instructions_text = CONDITIONS[experiment.condition_name]["block_instructions"][block_instruction_html_key];
    }
  }
  else if (EXPERIMENTS[experiment_name]["block_instructions"]) {
    if (EXPERIMENTS[experiment_name]["block_instructions"][block_instruction_html_key]) {
      block_instructions_text = EXPERIMENTS[experiment_name]["block_instructions"][block_instruction_html_key];
    }
  }

  // Get image names to replace (if they exist)
  if (CONDITIONS[experiment.condition_name]["block_instructions"]) {
    if (CONDITIONS[experiment.condition_name]["block_instructions"][block_instruction_image_key]) {
      block_instructions_images = CONDITIONS[experiment.condition_name]["block_instructions"][block_instruction_image_key];
    }
  }
  else if (EXPERIMENTS[experiment_name]["block_instructions"]) {
    if (EXPERIMENTS[experiment_name]["block_instructions"][block_instruction_image_key]) {
      block_instructions_images = EXPERIMENTS[experiment_name]["block_instructions"][block_instruction_image_key];
    }
  }

  // Replace images if any
  if (block_instructions_images != null) {
    return replace_trial_images(block_instructions_text, block_instructions_images);
  }

  if (block_instructions_text != null) {
    return block_instructions_text;
  }
  // stub jsPsych trial for no block instructions case
  return { type: "instructions" }
}

/**
 * Replaces the image text in the jsPsych instruction trial html with the given image paths
 * make sure if you have multiple instructions, the images corresponding to the instructions have the same name, 
 * plus naming convention: <instruction_name>_html, <instruction_name>_images
 * 
 * @param {jsPsych trial} instruction_trial jsPsych instruction trial
 * @param {array} images array of image paths
 * @return {jsPsych trial}
 */
function replace_trial_images(instruction_trial, images) {
  for (let i = 0; i < instruction_trial.pages.length; i++) {
    instruction_trial.pages[i] = get_html_with_images(instruction_trial.pages[i], images)
  }
  return instruction_trial;
}

/**
 * Checks if there is a block specific instruction image.
 * @param {object} experiment
 * @return {boolean}
 */
function is_block_specific_image(experiment) {
  let condition_name = experiment.condition_name;
  let first_block = get_first_block_name(experiment);
  let experiment_name = experiment.constructor.name.toLowerCase();
  if (experiment_name == "hybrid") {
    experiment_name = experiment.current_experiment.toLowerCase();
  }

  if (CONDITIONS[condition_name].instructions &&
    CONDITIONS[condition_name].instructions[experiment_name].custom_block_images &&
    CONDITIONS[condition_name].instructions[experiment_name].custom_block_images[`block_${first_block}_images`]) {
    return true;
  } else {
    return false;
  }
}

/**
 * Returns the name of the first block in the experiment
 * @param {object} experiment
 * @return {boolean}
 */
function get_first_block_name(experiment) {
  let experiment_name = experiment.constructor.name.toLowerCase();
  let constants = null;
  if (experiment_name == "hybrid") {
    let current_exp = `${experiment.current_experiment.toLowerCase()}_constants`;
    constants = experiment[current_exp][0];
  } else {
    constants = experiment.sub_conditions_constants[0];
  }
  let first_block = constants.block;

  //check if condition is multi_size_scatter_combined. If so, remove decimal in block name
  if (experiment.condition_name == "multi_size_scatter_combined") {
    first_block = first_block.replace(/\./g, '');
  }

  return first_block;
}
