const EXPERIMENTS_WITH_DIRECT_DATA = ["Estimation", "Ensemble_Average"]

/**
 * Retrieves the data for the corresponding experiment object.
 *
 * @param  {object}  experiment       Model object of the experiment 
 *
 * @return {assoc[]} dataset
 */
function get_data(experiment){
  var dataset;

  var trial_structure = experiment.trial_structure;
  var graph_type = experiment.graph_type;
  var condition = experiment.condition_name;
  var experiment_name = experiment.constructor.name;

  // Condition Data is Foundational Data from Experiment
  let condition_name_keys = condition.split("_");
  if (condition_name_keys.includes("base")){
    if (!EXPERIMENT_BASES[experiment_name][trial_structure]) {
      throw new Error ("Base for " + condition + " does not exist.");
    }
    dataset = EXPERIMENT_BASES[experiment_name][trial_structure];
    check_dataset(experiment_name, graph_type, dataset)
    return dataset;
  }
  // Condition Data for Experiment with no Base
  if(EXPERIMENTS_WITH_DIRECT_DATA[experiment_name]){
    if (!EXPERIMENT_CONDITIONS[experiment_name][condition]){
      throw new Error (condition + " is not supported.");
    }
  
    dataset = EXPERIMENT_CONDITIONS[experiment_name][condition];
    check_dataset(experiment_name, graph_type, dataset)
    return dataset;
  }
  // Data generated from Custom Case
  if(CUSTOM_TRIAL_STRUCTURE_CONDITIONS[trial_structure].includes(condition)){
    dataset = get_custom_subcondition(experiment);
    check_dataset(experiment_name, graph_type, dataset)
    return dataset;
  }
  // Data from Trial Structure
  let subconditions = EXPERIMENT_CONDITIONS[experiment_name][condition];
  switch(trial_structure){    
    case "foundational":
      check_subcondition_data(experiment_name,condition,trial_structure);
      check_trial_structure(experiment_name,trial_structure)
      dataset = create_foundational_condition_dataset(EXPERIMENT_BASES[experiment_name][trial_structure], subconditions);
      break;
    case "design":
      check_trial_structure(experiment_name,trial_structure)
      check_subcondition_data(experiment_name,condition,trial_structure);
      dataset = create_design_condition_dataset(EXPERIMENT_BASES[experiment_name][trial_structure], subconditions);
      break;
    case "custom":
      dataset = get_custom_subcondition(experiment);
      break;
    default:
      throw new Error(`Trial Structure ${trial_structure} not supported.`)
  }
  check_dataset(experiment_name, graph_type, dataset)

  return dataset;
}

/**
 * Checks if all subconditions have the necessary attributes to run the experiment.
 *
 * @param  experiment      {string}                 type of experiment
 *         graph_type.     {string}                 type of graph
 *         dataset         {array of JS objects}    dataset to check            
 */
function check_dataset(experiment_name, graph_type, dataset) {

  let name = "";
  let graph_attributes = GRAPH_TYPES[graph_type]["attributes"];
  let exp_attributes = EXPERIMENTS[experiment_name.toLowerCase()]["attributes"];
  let dist_attributes = DISTRIBUTION_TYPES["coordinate_distribution"]["attributes"];
  let graph_feature_attributes = {};


  let all_attributes = Object.assign({}, graph_attributes);
  all_attributes = Object.assign(all_attributes, exp_attributes);
  all_attributes = Object.assign(all_attributes, dist_attributes)
  for(let key in GRAPH_FEATURES){
    const feature_attributes = GRAPH_FEATURES[key]["attributes"];
    all_attributes = Object.assign(all_attributes, feature_attributes)
  }

  for (let subcondition of dataset) {

    // Check that each subcond has the necessary attributes to run the experiment
    for (let attribute_key in EXPERIMENTS[experiment_name.toLowerCase()]["attributes"]){

      let attribute = EXPERIMENTS[experiment_name.toLowerCase()]["attributes"][attribute_key];

      // Check that conditions has all required attributes
      if (subcondition[attribute_key] === undefined && attribute["required"]){
        throw new Error("Required attribute " + attribute_key + " was not found in the subcondition " + JSON.stringify(subcondition));
      }
    }

    for (let subcondition_key in subcondition) {

      // Check that all attributes have been defined in either experiment or graphing config
      if (all_attributes[subcondition_key] === undefined) {
        throw new Error("Attribute " + subcondition_key + " has not been defined in config files.");
      }

      // Check that if there is a valid_input, it is listed in the configs
      if (Array.isArray(subcondition[subcondition_key])) { //If array, loop through values and check each

        for (let value of subcondition[subcondition_key]) {
          if (all_attributes[subcondition_key].valid_inputs 
            && !all_attributes[subcondition_key].valid_inputs.includes(value)){
            throw new Error("Attribute " + subcondition_key + " has value " + value + " that is not a valid value. Valid values are: " + all_attributes[subcondition_key].valid_inputs);
          }
        }

      } else {
        if (all_attributes[subcondition_key].valid_inputs 
          && !all_attributes[subcondition_key].valid_inputs.includes(subcondition[subcondition_key])){
          throw new Error("Attribute " + subcondition_key + " has value " + subcondition[subcondition_key] + " that is not a valid value. Valid values are: " + all_attributes[subcondition_key].valid_inputs);
        }
      }
    }
  }
}

function check_subcondition_data(experiment_name,condition){
  if (!EXPERIMENT_CONDITIONS[experiment_name][condition]){
    throw new Error (condition + " is not supported.");
  }    
}

function check_trial_structure(experiment_name,trial_structure){
  if(!EXPERIMENT_BASES[experiment_name][trial_structure]){
    throw new Error ( trial_structure + " is not supported.");
  }
}

/**
 * Retrieves a smaller dataset (4 subconditions) given experiment, trial structure and condition.
 *
 * @param {string} experiment experiment name
 * @param {string} trial_structure "foundational","design","custom"
 * @param {string} condition condition name
 * @returns {assoc[]} dataset
 */
function get_data_subset(experiment, trial_structure, condition) {

  var dataset = get_data(experiment, trial_structure, condition);

  return dataset.slice(0, 4);
}

/**
 * Creates non-base Foundational Condition Datasets. Condition Data must be
 * the same length as Foundational Data.
 * 
 * @param {assoc} base_data 
 * @param {assoc} condition_data 
 * @returns {assoc} dataset
 */
function create_foundational_condition_dataset(base_data, condition_data){
  let base_length = base_data.length;
  let condition_length = condition_data.length;
  if (base_length != condition_length){
    throw Error("Foundational Condition Data Length is not the same as Foundational Base Data Length");
  }
  let condition_dataset = [];
  for(let i = 0; i < base_length; i++){
    let curr_subcondition_data = condition_data[i];
    let curr_base_data = base_data[i];
    let merged_data = Object.assign({},curr_base_data,curr_subcondition_data)
    condition_dataset.push(merged_data);
  }
  return condition_dataset;
}

/**
 * Takes the base Design data and appends the Condition data to create design x block
 * Subcondition array. Total number of subconditions is base length x number of blocks.
 * Attributes redefined in the blocked data overwrite those in base data.
 * 
 * @param {assoc[]} base_data 
 * @param {assoc[]} blocked_condition_data 
 * @returns {assoc[]} blocked_condition_dataset
 */
function create_design_condition_dataset(base_data, blocked_condition_data){
  let blocked_condition_dataset = [];
  let base_length = base_data.length;
  let number_of_blocks = blocked_condition_data.length;

  for(let i = 0; i < number_of_blocks; i++){
    for(let j = 0; j < base_length; j++){
      let subcondition = Object.assign({}, base_data[j], blocked_condition_data[i]);
      blocked_condition_dataset.push(subcondition);
    }
  }
  return blocked_condition_dataset;
}
