/**
 * FEEDBACK Green check mark or red X for right/wrong answers
 * @param {experiment} sequencing_symbols_exp 
 * @returns {trial}
 */
export function generate_feedback_screen(sequencing_symbols_exp){
    let feedback = {
        type: 'html-keyboard-response',
        choices: ['q'],
        post_trial_gap: 300,
        data: {type: 'feedback'},
        trial_duration: 500,
        on_start: function(feedback) {
          if (sequencing_symbols_exp.auto_mode) {
            feedback.trial_duration = 0;
          }
        },
        stimulus: function(){
      
          document.body.style.backgroundColor = sequencing_symbols_exp.trial_data.feedback_background_color;
      
          var last_trial_data = correlation_dependencies.get_last_trial_data(sequencing_symbols_exp.experiment_name)
          //console.log(last_trial_data)
          var last_trial_correct = last_trial_data.correct
      
          // For debugging purposes:
          if (last_trial_correct == -1){
            return '<p>' +
                   '<font style="font-size:50px; color:blue">Exiting from experiment.<p></font>'
          }
          else if (last_trial_correct == -2){
            return '<p>' +
                   '<font style="font-size:50px; color:blue">Skipping from subcondtion.<p></font>'
          }
      
          else if (last_trial_correct){
            return `<div align = "center">
                    <p>
                      <i class="fa fa-check-circle" style="font-size:50px; color:green; margin-right: 10px;"></i>
                      <font style="font-size:50px; color:green">
                      Correct!
                      </font>
                    <p>
                    <br>
                    </div>`
          }
          else{
            return `<div align = "center">
                    <p>
                      <i class="fa fa-close" style="font-size:50px; color:red; margin-right: 10px;"></i>
                      <font style="font-size:50px; color:red;"">
                      Incorrect!
                      </font>
                    <p>
                    <br>
                    </div>`
          }
        }
      };
    return feedback;
}

/**
 * Trial screen showing "end of practice" message
 * @param {experiment} sequencing_symbols_exp 
 * @returns {trial}
 */
export function generate_stop_practice_trials(sequencing_symbols_exp){
    let stop = {
        type: 'html-keyboard-response',
        stimulus: "<div align = 'center'> <font size = 20><p>This concludes the practice trials.<p>" + "<br><br><p><b>Any questions?</b></p></font></div>",
        data: {type: 'instruction'},
        on_start: function(stop){
          // Reset background color to feedback
          document.body.style.backgroundColor = sequencing_symbols_exp.trial_data.feedback_background_color;
          // skip if automation mode is turned on
          if (sequencing_symbols_exp.auto_mode) {
            stop.trial_duration = 0;
          }
        }
      };
      
    let stop_trials = {
        timeline: [stop]
    };
    return stop_trials;
}  

/**
 * This is a conditional block that will execute only if
 * the subcondition has a subcondition_instructions attribute
 * and it is the start of the subcondition.
 * @param {experiment} sequencing_symbols_exp 
 * @returns {trial}
 */
export function generate_practice_instruction_block(sequencing_symbols_exp){
  let block_type = "practice";
  let subcond_practice_instruction = {
    type: 'html-keyboard-response',
    stimulus: function(){

      let index = correlation_dependencies.get_subcondition_index(sequencing_symbols_exp, block_type);
      let constants = correlation_dependencies.get_constants(sequencing_symbols_exp, block_type, index);

      return constants.subcondition_instructions;
    }
  }

  let subcond_practice_instruction_block = {
    timeline: [subcond_practice_instruction],
    conditional_function: function() {
      let index = correlation_dependencies.get_subcondition_index(sequencing_symbols_exp, block_type);
      let constants = correlation_dependencies.get_constants(sequencing_symbols_exp, block_type, index);
      if (constants.subcondition_instructions && 
          correlation_dependencies.get_adjusted_quantity_matrix(sequencing_symbols_exp,block_type)[0].length == 0 && 
          correlation_dependencies.is_start_of_block(sequencing_symbols_exp, block_type)){
        return true;
      } else {
        return false;
      }
    }
  }
  return subcond_practice_instruction_block;
}

/**
 * Conditional looping logic timeline for experiment practice
 * Uses similar logic to normal experiment loop.
 * Sets and resets subcondition index flags in the experiment
 * @param {trial} subcond_practice_instruction_block 
 * @param {experiment} sequencing_symbols_exp 
 * @param {trial} practice_sequencing 
 * @param {trial} feedback 
 * @returns {trial}
 */
export function generate_practice_block(subcond_practice_instruction_block, sequencing_symbols_exp, practice_sequencing, feedback){
  let block_type = "practice";  
  let practice = {
        timeline: [subcond_practice_instruction_block, ...practice_sequencing, feedback], // We use same feedback block as that used in practice
        loop_function: function(data){ // Return true if timeline should continue
                                      // Return false if timeline should end
          let key_presses = [];
          data.values().forEach(function(elem){
                                key_presses.push(elem.key_press);
                                });
    
          // For debugging, if you want to exit out of experiment, press q:
          if (key_presses.includes(jsPsych.pluginAPI.convertKeyCharacterToKeyCode('q'))){
            // Turn flag on
            //sequencing_symbols_exp.first_trial_of_sub_condition = true;
            sequencing_symbols_exp.converged = false;
            sequencing_symbols_exp.end_sub_condition_debug = false;
            sequencing_symbols_exp.practice_condition_ended = true;
            return false;
          }
    
          if(correlation_dependencies.is_end_of_sub_condition(sequencing_symbols_exp,block_type)){
            // End practice
            //sequencing_symbols_exp.first_trial_of_sub_condition = true;
            sequencing_symbols_exp.converged = false;
            sequencing_symbols_exp.end_sub_condition_debug = false;
    
            // If there are still more subconditions, increment current index
            if (sequencing_symbols_exp.current_practice_condition_index < (sequencing_symbols_exp.sub_conditions_constants.length-1)){
              sequencing_symbols_exp.current_practice_condition_index++;
              console.log("!!!!!!!!!! Moved to new sub condition at index " + correlation_dependencies.get_subcondition_index(sequencing_symbols_exp, block_type));
              return true;
            }
            // Else end experiment
            else {
              sequencing_symbols_exp.practice_condition_ended = true;
              return false;
            }
          }
          // Else continue w/ current subcondition:
          else {
            return true;
          }
        }
    };
    return practice;
}


/**
 * Generates the main experiment timline that handles looping logic that determines
 * subcondition start and ends, subcondition index, as well as if the experiment should conclude or not
 * Sets the relevant experiment flags in the given experiment for subconditon index logic
 * 
 * @param {experiment} sequencing_symbols_exp 
 * @param {trial} subcond_instruction_block 
 * @param {trial} experiment_trials trials generated by experiment's generate_trial function
 * @param {trial} feedback 
 * @returns {trial} 
 */
export function generate_experiment(sequencing_symbols_exp, subcond_instruction_block, experiment_trials, feedback){
  let block_type = "test";
  let experiment = {
    timeline: [subcond_instruction_block, ...experiment_trials, feedback],
    loop_function: function(data){ // Return true if timeline should continue
                                   // Return false if timeline should end
  
      let key_presses = [];
      data.values().forEach(function(elem){
                            key_presses.push(elem.key_press);
                            });
                            
      // For debugging, if you want to exit out of experiment, press q:
      // TODO rework this flag
      if (key_presses.includes(jsPsych.pluginAPI.convertKeyCharacterToKeyCode('q'))){
        return false;
      }
      // If subcondition should end:
      if(correlation_dependencies.is_end_of_sub_condition(sequencing_symbols_exp, block_type)){
        //sequencing_symbols_exp.first_trial_of_sub_condition = true;
        sequencing_symbols_exp.converged = false;
        sequencing_symbols_exp.end_sub_condition_debug = false;
  
        // If there are still more subconditions, increment current index
        if (sequencing_symbols_exp.current_sub_condition_index < (sequencing_symbols_exp.sub_conditions_constants.length-1)){
          sequencing_symbols_exp.current_sub_condition_index++;
          console.log("!!!!!!!!!! Moved to new sub condition at index " + sequencing_symbols_exp.current_sub_condition_index);
          return true;
        }
        // Else end experiment
        else {
          return false;
        }
      }
      // Else continue w/ current subcondition:
      else {
        return true;
      }
    },
    on_finish: function(data) {
      sequencing_symbols_exp.trial_data = data;
    }
  };
  return experiment;
};


/**

 */
export function generate_experiment_end(sequencing_symbols_exp){

        stimulus: '<div align = "center">' +
                  '<p><font size = 10>You have completed the experiment!<p></font>' +
                  '</div>' ,
        on_start: function(){
      
          sequencing_symbols_exp.export_trial_data();
          //sequencing_symbols_exp.export_summary_data();
      
          // Reset background color to feedback
          document.body.style.backgroundColor = sequencing_symbols_exp.trial_data.feedback_background_color;
        }
      };
    return experiment_end;
}
