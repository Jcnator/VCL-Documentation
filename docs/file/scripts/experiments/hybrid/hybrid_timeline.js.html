// Stevens PRACTICE TIMELINE
if (isPractice == "true") {

  let sample_steven_trials = stevens_exp.prepare_sample_trials();
  let stevens_practice_trial = stevens_exp.generate_trial("practice");

  let stevens_practice_timeline = {
    timeline: [stevens_practice_trial],
    loop_function: function (data) {
      // For debugging, if you want to exit out of experiment, press q:
      if (jsPsych.pluginAPI.convertKeyCharacterToKeyCode('q') == data.values()[0].key_press) {
        stevens_exp.end_practice_experiment();
        stevens_exp.round_end = false;
        console.log("!!!!!!!!!! Practice trials finished ");
        return false;
      }

      if (stevens_exp.end_round()) {
        console.log("Ending Round")

        // Save the midpoint for exclusion criteria calculations later
        let curr_index = stevens_exp.curr_practice_condition_index;
        stevens_exp.practice_trial_data[curr_index].push(stevens_exp.trial_data);
        console.log("!!!!!!!!!!!!!!PRACTICE TRIAL DATA")
        console.log(stevens_exp.practice_trial_data[curr_index])
        // If subcondtion should end
        if (stevens_exp.end_sub_condition("practice", curr_index)) {
          // Check if there is another subcondition and increase the subcondition index
          if (stevens_exp.curr_practice_condition_index < (stevens_exp.practice_conditions_constants.length - 1)) {
            console.log("!!!!!!!!!! Moved to new sub condition at index " + stevens_exp.curr_practice_condition_index + 1);
          }
          else {
            // If there are no more subconditions, end the practice trials
            console.log("!!!!!!!!!! Practice trials finished ");
            stevens_exp.end_practice_experiment();
            stevens_exp.round_end = false;
          }
          // Practice Timeline End. Goes to Practice End Trial to check exclusion criteria.
          return false;
        }
        console.log("!!!!!!!! GO TO NEXT ROUND ");
        stevens_exp.round_end = true;
        return true;
      }
      // Else continue w/ current round:
      return true;
    }
  };

  let stop_stevens_practice = {
    type: 'html-keyboard-response',
    stimulus: "",
    on_start: function (stop_stevens_practice) {
      // Reset background color to feedback
      document.body.style.backgroundColor = 'WHITE';

      let practice_condition_data = stevens_exp.practice_trial_data[stevens_exp.curr_practice_condition_index];
      let range = stevens_exp.calculate_exclusion_criteria_for_subcondition(practice_condition_data);
      let results = `<div align = "center" style = "text-align: left; float:left;"> <font size = 6>
                    Range: ${range}
                    <br>
                    </font>
                    </div>`

      let current_exlcusion = range < STEVENS_EXCLUSION_VALUE;
      if (hybrid_exp.is_below_exclusion_0 == null) {
        hybrid_exp.is_below_exclusion_0 = current_exlcusion;
        // Test first practice sub condition
        if (hybrid_exp.is_below_exclusion_0 && hybrid_exp.is_below_exclusion_1 == null) {
          stevens_exp.num_trial_complete = 0;
          stop_stevens_practice.stimulus = "<div align = 'center'>" + results + "</div>" +
            "<div align = 'center'> <font size = 6><br><br><p> This concludes the Stevens Task practice trials.</p>" +
            "<br><br><p><b>Any questions?</b></p></font></div>";
          stevens_exp.end_practice_experiment();
        }
        else {
          // Increase index here to avoid calculate_exclusion_criteria_for_subcondition() error
          stevens_exp.curr_practice_condition_index++;
          stop_stevens_practice.stimulus = "<div align = 'center'>" + results + "</div>" +
            "<div align = 'center'> <font size = 6><br><br><p> We will be doing one more practice round.</p>" +
            "<br><br><p><b>Any questions?</b></p></font></div>";
        }
      }
      else {
        hybrid_exp.is_below_exclusion_1 = current_exlcusion;
        // Test for the second practice sub condition
        if (hybrid_exp.is_below_exclusion_1) {
          stop_stevens_practice.stimulus = "<div align = 'center'>" + results + "</div>" +
            "<div align = 'center'> <font size = 6><br><br><p> This concludes the Stevens Task practice trials. </p>" +
            "<br><br><p><b>Any questions?</b></p></font></div>";
          stevens_exp.end_practice_experiment();
        }
        else {
          stop_stevens_practice.stimulus = "<div align = 'center'>" + results + "</div>" +
            "<div align = 'center'> <font size = 6><br><br><p> You will be unable to participate in the experiment. Please exit the window.</p>";
          stop_stevens_practice.choices = jsPsych.NO_KEYS;
        }
      }
    },
    data: { type: 'instruction' }
  }

  let stevens_practice_timelines = {
    timeline: [stevens_practice_timeline, stop_stevens_practice],
    loop_function: function () {
      if (hybrid_exp.is_below_exclusion_0 || hybrid_exp.is_below_exclusion_1) {
        return false;
      }
      console.log("Repeat timeline")

      return true;
    }
  }

  timeline.push(stevens_practice_timelines);

}
