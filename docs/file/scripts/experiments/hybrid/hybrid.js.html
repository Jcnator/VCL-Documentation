    import { balance_subconditions } from '/scripts/experiment-properties/balancing/balancing_controller.js'
import { get_data } from '/scripts/experiment-properties/data/data_controller.js'
import { EXPERIMENTS } from "/config/experiments-config.js";

export default class Hybrid{

    constructor(params){
        console.log(params)
        let trial_structure = params["trial_structure"];
        let condition_name = params["condition"];
        let graph_type = params["graph_type"];
        let balancing_type = params["balancing"];
    
        this.condition_name = condition_name;
        this.subject_id = params["subject_id"];
        this.subject_initials = params["subject_initials"];
        this.auto_mode = false;
    
        this.is_practice_trials = params["is_practice_trials"];
        this.is_screening = params["is_screening"];
        this.is_local_save = params["is_local_save"]
        this.manual_scaling_factor = params["manual_scaling_factor"]
        this.custom_subcondition_order = params["custom_subcondition_order"];


        // ========================================
        // PARAMETER CHECKING

        // **NOTE: EXPERIMENTS variable comes from /public/config/experiments-config.js
        if (!EXPERIMENTS["hybrid"]["trial_structure"].includes(trial_structure)) {
            throw Error(trial_structure + " is not supported.");}
        else {
            this.trial_structure = trial_structure;
        }
    
        if (!EXPERIMENTS["hybrid"]["graph_type"].includes(graph_type)){
            throw Error(graph_type + " is not supported.")}
        else {
            this.graph_type = graph_type;
        };
    
        if (!EXPERIMENTS["hybrid"]["balancing_type"].includes(balancing_type)) {
            throw Error(balancing_type + " is not supported.") }
        else {
            this.balancing_type = balancing_type;
        }
       

        
        // Extract raw constants
        this.raw_constants = get_data(this);
        console.log(this.raw_constants)

        this.current_experiment = "Stevens";
        this.prepare_experiment()
        this.super_condition_constants = [];
    }
    

    prepare_experiment(){
        let dataset = this.raw_constants;
        this.sub_condition_order = balance_subconditions(this.balancing_type, this.constructor.name.toLowerCase(), dataset.length, dataset, "block", this.custom_subcondition_order);
        this.custom_order_string = 'br' + this.sub_condition_order.toString().replaceAll(',','');
    }

    /**
     * 
     * @returns {assoc[]}
     */
    prepare_hybrid_order(){
        let num_blocks = 5;
        let stevens_block_size = 2;
        let jnd_block_size = 3;
        let total_block_size = stevens_block_size + jnd_block_size;
        let hybrid_order_constants = []
        for(let i = 0; i < num_blocks; i++){
            let stevens_0_idx = i * stevens_block_size;
            let stevens_1_idx = stevens_0_idx + 1;
            let jnd_0_idx = i * jnd_block_size;
            let jnd_1_idx = jnd_0_idx + 1;
            let jnd_2_idx = jnd_1_idx + 1;
            
            hybrid_order_constants.push(this.stevens_constants[stevens_0_idx])
            hybrid_order_constants.push(this.jnd_constants[jnd_0_idx])
            hybrid_order_constants.push(this.jnd_constants[jnd_1_idx])
            hybrid_order_constants.push(this.jnd_constants[jnd_2_idx])
            hybrid_order_constants.push(this.stevens_constants[stevens_1_idx])
        }

        this.super_condition_constants = hybrid_order_constants;
    }

    /**
     * When called, will save summary data and individual trial data into firebase.
     * Parameters are passed on the timeline
     */
    export_trial_data(jnd_summary_data, stevens_summary_data) {
        let summary_data = [];

        let screening_json = jsPsych.data.get().filter({trial_type: 'survey-html-form'}).select('responses').values[0]
        // Participant information
        // indexing might need to be genralized better if there are previous trials before the start of the experiment
        if(screening_json){
            console.log('Saving Screening')
            let screening_responses = JSON.parse(screening_json)
            let screening_data = {};
            screening_data.education = screening_responses[2].value
            screening_data.stat_classes_take = screening_responses[3].value
            screening_data.vision = screening_responses[4].value
            screening_data.gender = screening_responses[5].value
            screening_data.age = screening_responses[6].value
            summary_data.push(screening_data);
        }
        //Get experiment's entire trial data 
        let trial_data = this.get_trial_data();
        summary_data = summary_data.concat(stevens_summary_data);
        summary_data = summary_data.concat(jnd_summary_data);
        console.log(summary_data)

        if(this.is_local_save === "true"){
            trial_data.localSave('csv', 'hybrid_trial_data.csv');
            // TODO fix the summary data to csv
            // jnd_summary_data.localSave('csv', 'jnd_summary_data.csv')
            // stevens_summary_data.localSave('csv', 'stevens_summary_data.csv')    
        }
        else{
            let json_trial_data = JSON.parse(trial_data.json());
            var data = summary_data.concat(json_trial_data);
            data = JSON.stringify(data);
            // Firebase upload
            // Sends the data to the app.js listener
            $.post("/api/experiment_data",
                    {
                        trial_data: json_trial_data, 
                        summary_data: summary_data, 
                        experiment: "hybrid", 
                        condition: params["condition"], 
                        prolific_pid: params["subject_id"], 
                        study_id: params["study_id"], 
                        session_id: params["session_id"],
                        subject_initials: params["subject_initials"]
                    },
                    function(data, status) {
                        console.log("Data: " + data + "\nStatus: " + status)
                    }
                );
        }
    }

    // Getters

    /**
     * Gets jsPsych trial with given filters. Only returns 'blank'
     * type trials since those are the only ones that save data.
     * 
     * @param   {assoc array} filters 
     * @returns {jsPsych data} entire jsPsych data object. Needs to be furtherly filtered later to be useful
     */
    get_jsPsych_data(filters){
        return jsPsych.data.get().filter(filters)
                                .filterCustom(function(x) {
                                    return (x.correct != -1);
                                });
    }

    /**
     * Returns the data of the last trial on the jsPsych data object
     * @returns {assoc array} last jsPsych trial data object
     */
    get_last_trial_data(experiment){
        return jsPsych.data.get().filter({type: experiment}).last(1).values()[0];
    }

    /**
     * Returns releveant array of the experiment's trial data (in order from first to last)
     * Non-relevant fields are ignored.
     * "Column order" if downloaded to csv is "randomized" because trial data object is an unordered map
     * 
     * @returns {array} [{trial_0},{trial_1},...,{trial_n}]
     */
    get_trial_data(){
        return jsPsych.data.get().filter({run_type: 'test'})
                                    .filterCustom(function(x){ //Don't include the exit trials
                                        return x.correct != -1;
                                    })
                                    .filterCustom(function(x){
                                        if (x.type === "stevens" || x.type === "jnd")  // return only stevens and jnd trials
                                        return x;
                                    })
                                    .filterCustom(function(x){ //Don't include trials with no user input
                                        return x.rt != null;
                                        })
                                    .filterCustom(function(x){ // Only include the round end steven trials
                                        if(x.type==="stevens"){
                                            return x.trial_end == 1;
                                        }
                                        return x
                                        })
                                    // Steven's trial variables
                                    .ignore('run_type')
                                    .ignore('left_correlation')
                                    .ignore('right_correlation')
                                    .ignore('round_refreshes')
                                    .ignore('high_ref_is_right')
                                    // JND's trial variables
                                    .ignore('type')
                                    .ignore('run_type')
                                    .ignore('left_correlation')
                                    .ignore('right_correlation')
                                    // Experiment constants
                                    .ignore('style_attributes')
                                    .ignore('SD')
                                    .ignore('background_color')
                                    .ignore('feedback_background_color')
                                    .ignore('initial_difference')
                                    .ignore('mean')
                                    .ignore('num_SD')
                                    // These are variables forced on by jsPsych
                                    .ignore('stimulus')
                                    .ignore('key_press')
                                    .ignore('choices')
                                    .ignore('trial_type')
                                    .ignore('trial_index')
                                    .ignore('time_elapsed')
                                    .ignore('internal_node_id');
        } 

    /**
     * Get's experiment relevant summary data.
     * Relevant summary data includes: independent variables, overall JND, convergance JND (last 24), convergance results, trial count
     * 
     * @returns {assoc[]} array of summary data
     */
    get_summary_data(){
        let summary_data = [];
        let screening_json = jsPsych.data.get().filter({trial_type: 'survey-html-form'}).select('responses').values[0]
        // Participant information
        // indexing might need to be genralized better if there are previous trials before the start of the experiment
        if(screening_json){
            console.log('Saving Screening')
            let screening_responses = JSON.parse(screening_json)
            let screening_data = {};
            screening_data.education = screening_responses[2].value
            screening_data.stat_classes_take = screening_responses[3].value
            screening_data.vision = screening_responses[4].value
            screening_data.gender = screening_responses[5].value
            screening_data.age = screening_responses[6].value
            summary_data.push(screening_data);
            }

        stevens_summary = [];


        // Subcondition Data
        for (let i = 0; i<this.super_condition_constants.length; i++) {
            let summary_data_sub;


            summary_data.push(summary_data_sub)
        }

        return summary_data;
    }
}
